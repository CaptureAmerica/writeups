#!/usr/bin/env python
# -*- coding:utf-8 -*-
from PIL import Image

MAX=29
f=open("STEG05.csv","r").read().split("\n")
a=[]
for line in f:
    for y in range(MAX):
        val = line[y]
        if val=='0':
            a.append((255,255,255))
        else:
            a.append((0,0,0))

im = Image.new("RGB",(MAX,MAX))
im.putdata(a)
im.save("QR.png")
